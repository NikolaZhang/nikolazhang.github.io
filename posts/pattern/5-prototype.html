    @Override
    public Prototype cloneObject() {
        ByteArrayOutputStream bos = new ByteArrayOutputStream();
        try (ObjectOutputStream oos = new ObjectOutputStream(bos)) {
            oos.writeObject(this);
            oos.flush();
        } catch (IOException e) {
            e.printStackTrace();
            return null;
        }
        // ä»å­—èŠ‚æµä¸­è¯»å–å¯¹è±¡

            e.printStackTrace();
            return null;
        }
    }

}
